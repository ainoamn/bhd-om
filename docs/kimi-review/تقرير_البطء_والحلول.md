## 3. الحلول

### الحل 1: تخزين HTML مؤقتاً في الذاكرة (أسرع حل)

**الملف:** `app/api/admin/legacy-real-estate/[[...path]]/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { requireAuth } from '@/lib/auth/guard';
import { buildLegacyBridgeMinimalPayload, resolveLegacyBridgeLocale } from '@/lib/server/legacyBridge';
import { contentTypeForLegacyFile, readLegacyRealEstateFile, readLegacyRealEstateHtmlWithBridge } from '@/lib/server/legacyRealEstateFiles';
import { isAdminLikeRole } from '@/lib/auth/roles';

export const dynamic = 'force-dynamic';

// ========== CACHE جديد ==========
let cachedHtml: Buffer | null = null;
let cachedHtmlMtime: number = 0;

async function getCachedHtml(segments: string[] | undefined, embeddedPayload?: any): Promise<Buffer> {
  const fs = await import('fs/promises');
  const path = await import('path');
  const LEGACY_ROOT = path.join(process.cwd(), 'legacy', 'bhd-real-estate');
  const fileName = segments?.length ? segments[segments.length - 1]! : 'bhd-real-estate.html';
  const filePath = path.join(LEGACY_ROOT, ...segments || ['bhd-real-estate.html']);
  
  // فحص آخر تعديل للملف
  const stats = await fs.stat(filePath);
  const mtime = stats.mtimeMs;
  
  // إذا الملف لم يتغير و cache موجود → ارجع الـ cache
  if (cachedHtml && mtime === cachedHtmlMtime && !embeddedPayload) {
    return cachedHtml;
  }
  
  // قراءة جديدة
  const raw = await readLegacyRealEstateFile(segments);
  const html = readLegacyRealEstateHtmlWithBridge ? 
    await readLegacyRealEstateHtmlWithBridge(segments, embeddedPayload) : raw;
  
  // حفظ في cache
  if (!embeddedPayload) {
    cachedHtml = html;
    cachedHtmlMtime = mtime;
  }
  
  return html;
}
// ========== نهاية CACHE ==========

type RouteContext = { params: Promise<{ path?: string[] }> };

export async function GET(req: NextRequest, context: RouteContext) {
  try {
    const auth = await requireAuth(req);
    if (auth instanceof NextResponse) return auth;
    
    const roleOk = isAdminLikeRole(auth.role) || auth.role === 'ADMIN' || auth.role === 'SUPER_ADMIN';
    if (!roleOk) return NextResponse.json({ error: 'Forbidden' }, { status: 403 });
    
    const userId = auth.userId;
    if (!userId) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

    const locale = resolveLegacyBridgeLocale(req);
    const { path: segments } = await context.params;
    const fileName = segments?.length ? segments[segments.length - 1]! : 'bhd-real-estate.html';
    const isMainHtml = fileName === 'bhd-real-estate.html' || fileName.endsWith('.html');

    let body: Buffer;
    let bridgeStatus = 'none';
    
    if (isMainHtml) {
      const embedded = await buildLegacyBridgeMinimalPayload(userId, locale);
      bridgeStatus = embedded ? 'embedded' : 'empty';
      // استخدام CACHE
      body = await getCachedHtml(segments, embedded);
    } else {
      body = await getCachedHtml(segments);  // ← استخدام CACHE
    }

    return new NextResponse(new Uint8Array(body), {
      status: 200,
      headers: {
        'Content-Type': contentTypeForLegacyFile(fileName),
        'Cache-Control': 'private, max-age=300, stale-while-revalidate=60',  // ← Cache لـ 5 دقائق
        'X-Bhd-Bridge': bridgeStatus,
      },
    });
  } catch (error) {
    // ...
  }
}
```

**التأثير:** تقليل من 500ms-2s إلى 1-5ms (قراءة من الذاكرة)

### الحل 2: Cache `buildLegacyBridgeMinimalPayload`

**الملف:** `lib/server/legacyBridge.ts`

```typescript
// ========== CACHE جديد ==========
const bridgeCache = new Map<string, { payload: LegacyBridgePayload; expiresAt: number }>();
const BRIDGE_CACHE_TTL = 60_000; // 60 ثانية

export async function buildLegacyBridgeMinimalPayload(
  authUserId: string,
  locale: 'ar' | 'en' = 'ar'
): Promise<LegacyBridgePayload | null> {
  // فحص cache
  const cached = bridgeCache.get(authUserId);
  if (cached && cached.expiresAt > Date.now()) {
    // تحديث الوقت فقط
    return {
      ...cached.payload,
      syncedAt: new Date().toISOString(),
    };
  }
  
  // البناء العادي...
  const currentDbUser = await prisma.user.findUnique({...});
  if (!currentDbUser) return null;
  const currentUser = mapSiteUserToLegacyUser(currentDbUser);
  
  // تجاوز contractLifecycle إذا كان في cache
  let contractLifecycle = contractLifecycleCache?.data;
  if (!contractLifecycle) {
    try {
      const canon = await loadCanonicalContractStatusesFromNeon(true);
      contractLifecycle = { statuses: canon.statuses, byUnit: canon.byUnit, ... };
    } catch (e) { ... }
  }
  
  const payload = { ... };
  
  // حفظ في cache
  bridgeCache.set(authUserId, { payload, expiresAt: Date.now() + BRIDGE_CACHE_TTL });
  
  return payload;
}
// ========== نهاية CACHE ==========
```

**التأثير:** تقليل من 7-15s إلى 200ms-500ms

### الحل 3: Cache `loadCanonicalContractStatusesFromNeon`

**الملف:** `lib/server/contractLifecycle.ts`

```typescript
// ========== CACHE جديد ==========
let contractLifecycleCache: {
  data: ContractStatusMaps & { groupsProcessed: number };
  timestamp: number;
} | null = null;
const CONTRACT_CACHE_TTL = 300_000; // 5 دقائق

export async function loadCanonicalContractStatusesFromNeon(persist: boolean) {
  // فحص cache أولاً
  if (contractLifecycleCache && (Date.now() - contractLifecycleCache.timestamp) < CONTRACT_CACHE_TTL) {
    return { ...contractLifecycleCache.data, persisted: false };
  }
  
  // البناء العادي...
  const result = await reconcileSavedContractsLifecycle(contractsRaw, accountingRaw);
  
  // حفظ في cache
  contractLifecycleCache = {
    data: { statuses: result.statuses, byUnit: result.byUnit, groupsProcessed: result.groupsProcessed },
    timestamp: Date.now(),
  };
  
  return { ...result, persisted };
}
// ========== نهاية CACHE ==========
```

**التأثير:** تقليل من 7-15s إلى 0ms (من cache)

    async start(controller) {
      // إرسال الـ head مباشرة

      controller.enqueue(htmlHead);
      
      // إرسال الباقي

      controller.enqueue(htmlRest);
      
      controller.close();
    },
  });
  
  return new NextResponse(stream, {
    headers: { 'Content-Type': 'text/html; charset=utf-8' },
  });
}
